#Change application name
spring.application.name=udemy.practice

# To enable the h2-console
spring.h2.console.enabled = true 

#rename the url
spring.datasource.url = jdbc:h2:mem:pogi 

#rename the user
spring.datasource.user=pogi 

#change password
spring.datasource.password=mypassword

#enable automatically restart
spring.devtools.restart.enabled=true

#delay the initialization of sql
spring.jpa.defer-datasource-initialization=true

#enable show sql in log
spring.jpa.show-sql=true 

#to show all erros
logging.level.org.springframework=debug

#This properties is for viewing jsp file (/src/main/resources/META-INF/resources/WEB-INF/jsp)
spring.mvc.view.prefix=/WEB-INF/jsp
spring.mvc.view.suffix=.jsp
